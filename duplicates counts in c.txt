#include<stdio.h>
int main()
{
    int n,i,j,count=0,arr[20];
    printf("enter number of elements");
    scanf("%d",&n);
    printf("array elements");
    for(i=0;i<n;i++)
    {
        scanf("%d",&arr[i]);
        
    }
for(i=0;i<n;i++)
{
    for(j=i+1;j<n;j++)
    {
        if(arr[i]==arr[j])
        count++;
        break;
    }
}
printf("total duplicates elements %d\n ",count);
return 0;
}
